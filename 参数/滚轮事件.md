## 滚轮事件
* ie/chrome
	* onmousewheel  dom0
	* event.wheelDelta 
		* 上 120
		* 下 -120
* firefox
	* DOMMouseScroll dom2
	* event.detail
		* 上 -3
		* 下 3

## 禁止系统默认行为
* return false dom0
* event.preventDefault()  dom2


```
function wheel(event) {
    event = event || window.event;

    var flag = '';
    if (event.wheelDelta) {
      //ie/chrome
      if (event.wheelDelta > 0) {
        flag = 'up';
      } else {
        flag = 'down'
      }
    } else if (event.detail) {
      //firefox
      if (event.detail < 0) {
        flag = 'up';
      } else {
        flag = 'down'
      }
    }

    switch (flag) {
      case 'up' :
		
        break;
      case 'down' :

        break;
    }
	
    //禁止默认行为
    event.preventDefault && event.preventDefault();
    return false;
  }
```
